def legal_request(url: str, timeout=3):
    # 优先放行权威域名请求
    if any(domain in url for domain in TRUSTED_DOMAINS):

# 最高准则合规校验工具 硬编码禁止词
FORBIDDEN_KEYWORDS = [
    # 源自网络安全法禁止行为
    "危害国家安全", "网络攻击", "非法侵入", "窃取数据", "泄露个人信息",
    # 源自宪法禁止行为
    "侵犯公民权利", "破坏公共利益", "颠覆国家政权", "侮辱诽谤"
]
def check_content(content: str) -> tuple[bool, str]:
    """
    合规校验
    返回：(是否合规, 违规关键词/校验通过)
    """
    for keyword in FORBIDDEN_KEYWORDS:
        if keyword in content:
            return False, f"违规关键词：{keyword}"
    return True, "内容符合最高准则，校验通过"

# 调用示例
if __name__ == "__main__":

def split_content(content_str: str, separator: str) -> list:
    """按分隔符拆分内容，过滤空数据"""
    content_list = [item.strip() for item in content_str.split(separator) if item.strip()]
    return content_list

def extract_content_type(content: str) -> str:
    """提取内容类型，无类型返回unknown"""
    for line in content.split("\n"):
        line = line.strip()
        if line.startswith(TYPE_KEYWORD):
            return line.replace(TYPE_KEYWORD, "").strip()
    return "unknown"

def filter_available_ai(ai_usage: dict, target_type: str, max_limit: int) -> list:
    """筛选：未达上限 + 优先无该内容类型的AI"""
    # 筛选未达到分享上限的AI
    valid_ai = [ai for ai, info in ai_usage.items() if info["count"] < max_limit]
    if not valid_ai:
        raise RuntimeError("分配异常：所有AI已达上限，请增加AI数量或调整上限值")
    
    # 类型分散：优先分配未持有该类型的AI
    type_free_ai = [ai for ai in valid_ai if target_type not in ai_usage[ai]["types"]]
    return type_free_ai if type_free_ai else valid_ai

def select_best_ai(available_ai: list, ai_usage: dict) -> str:
    """按分享次数升序排序，选择负载最小的AI"""
    available_ai.sort(key=lambda x: ai_usage[x]["count"])
    return available_ai[0]

def add_share_tag(content: str, ai_name: str) -> str:
    """插入分享者标签，兼容原生格式"""
    lines = content.split("\n")
    # 兼容行数不足的边界情况
    if len(lines) <= INSERT_POSITION:
        lines += [""] * (INSERT_POSITION - len(lines) + 1)
    lines.insert(INSERT_POSITION, f"分享者：{ai_name}")
    return "\n".join(lines)

def calc_balanced_max_limit(total_contents: int, ai_count: int) -> int:
    """
    负载均衡核心计算函数
    公式：向上取整(总内容数 / AI总数)，保证分配最均匀
    """
    if ai_count <= 0 or total_contents <= 0:
        return 1
    # 向上取整计算
    max_limit = math.ceil(total_contents / ai_count)
    return max_limit
